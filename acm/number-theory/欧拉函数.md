int phi(int n){
    int ans=n;
    for(int i=2;i*i<=n;i++)if(n%i==0){
        ans=ans/i*(i-1);
        while(n%i==0)n/=i;
    }
    if(n>1)ans=ans/n*(n-1);
    return ans;
}

#define N 100001
int e[N];
void Init(){
    for(int i=1;i<=N;i++)e[i]=i;
    for(int i=2;i<=N;i+=2)e[i]/=2;
    for(int i=3;i<=N;i+=2)
        if(e[i]==i){
            for(int j=i;j<=N;j+=i)
                e[j]=e[j]/i*(i-1);
        }
}
